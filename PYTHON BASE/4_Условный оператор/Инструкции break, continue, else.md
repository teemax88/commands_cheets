```python
# ctrl+c останавливаем бесконечный цикл
i = 1
while True:  # бесконечный цикл
    print("Итог итерации = ", i)
    if i == 10:  # закончит цикл (break) когда i = 10, т.е. перейдёт в 8-ю строку
        break
    i += 1
print("Hello")

while True:  # вводит от польз. строки и выводит их с их длинной пока не введено exit
    a = input()
    if a == 'exit':
        break
    if len(a) < 5:  # если длинна строки меньше 5-ти возврат в начало цикла (continue) без выполнения того что ниже
        continue
    print(a, len(a))

a = [12, 34, 56, 27]  # Yes если в списке все чётные, No если в списке есть нечётное
while len(a) > 0:
    last = a.pop()  # удалит последний элемент в списке перед этим присвоив его в last
    if last % 2 != 0:  # если попался нечётный - напечатает No и выйдет из цикла в 26-ю строку
        print("No")
        break
else:  # напечатает Yes если закончился предидущий цикл и не сработал на нечётность
    print("Yes")
```
