```python
"""
Класс Counter позволяет посчитать количество вхождения каждого элемента переданной коллекции
"""
from collections import Counter

s = 'abracadabra'
words = ['Donald', 'Mickey', 'Donald', 'Mickey', 'Mickey']
letters = Counter(s)
print(letters)  # Counter({'a': 5, 'b': 2, 'r': 2, 'c': 1, 'd': 1})
# вывод словаря в котором ключи-буквы, значения- сколько раз они встретились
names = Counter(words)
print(names)  # Counter({'Mickey': 3, 'Donald': 2})

# поскольку это словарь то мы можем обратиться к ключам или значениям
print(names['Donald'])  # 2 - значение ключа Donald

# у Counter есть специфические методы:
for i in names.elements():  # метод element выведет столько раз ключ сколько его значение
    print(i)
# Donald
# Donald
# Mickey
# Mickey
# Mickey

print(names.most_common())  # [('Mickey', 3), ('Donald', 2)] выведет сначала ключи с наибольшими ключами
print(letters.most_common())  # [('a', 5), ('b', 2), ('r', 2), ('c', 1), ('d', 1)] сначала ключи с наибольшими ключами
print(letters.most_common(3))  # [('a', 5), ('b', 2), ('r', 2)] 3-и значения, сначала ключи с наибольшими ключами

d = Counter()
for i in [1, 2, 3, 4, 5, 2, 3, 4, 4, 4, 1, 1, 1, 2, 3]:
    d[i] = d[i] + 1  # посчитает сколько раз встреться цифра
print(d)  # Counter({1: 4, 4: 4, 2: 3, 3: 3, 5: 1})

d_1 = Counter([2, 3, 4, 2, 3, 5, 5, 5, 1, 9])
print(d_1)  # Counter({5: 3, 2: 2, 3: 2, 4: 1, 1: 1, 9: 1})

print(d + d_1)  # Counter({1: 5, 2: 5, 3: 5, 4: 5, 5: 4, 9: 1}) сложение двух счётчиков
print(d_1 - d)  # Counter({5: 2, 9: 1}) вычитание счётчиков (останутся только положительные значения)

t = Counter()  # присвоили в т пустой счётчик (в котором могут быть только неизменяемые объекты)
t[1] = 5
print(t)  # Counter({1: 5})
t[(3, 5, 7)] = 13  # можно добавить в качестве ключа множество, но не список
print(t)  # Counter({(3, 5, 7): 13, 1: 5})
t[(3, 5, 7)] += 55  # увеличили значене ключа (3,5,7) на 55
print(t)  # Counter({(3, 5, 7): 68, 1: 5})

```
