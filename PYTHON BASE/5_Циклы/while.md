## Алгоритм Евклида (НОД, НОК)
```python
a, b = int(input()), int(input())  # найдём НОД (наибольший общий делитель)
# но при таком решении чесел например 10000000 и 5 будет очень много вычислений а ответ будет 5
while a != b:  # пока а != b, когда они равны это и есть НОД
    if a > b:  # если а > b вычитаем с большего меньшее
        a = a - b
    else:  # иначе b > a вычитаем с большего меньшее
        b = b - a55
print("НОД =", a)
```

```python
a, b = int(input()), int(input())  # найдём НОД (наибольший общий делитель) (например 100,5 или 5,100)
# но при таком решении чесел например 100 и 5 будет всего 2а вычисления а ответ будет 5
while b > 0:  # минимальный НОД равен 1
    a, b = b, a % b  # вместо строк 14-16(в а записать b, потом в b записать a%b
    # c = a % b  # в переменную с запись остатка от деления 100 % 5 = 0, наоборот 5 % 100 = 5
    # a = b  # запишем в а значение b (или 5 или 100 с примера)
    # b = c  # запишем в b значение c (или 0 или 5 с примера)
print("НОД =", a)
```

```python
# НОК (наименьшее общее кратное) = a*b/НОД
a, b = map(int, input().split())
a1, b1 = a, b
while b > 0:  # минимальный НОД равен 1
    a, b = b, a % b
print("НОК = ", int(a1 * b1 / a))
```

## Обход всех цифр числа
```python
x = int(input())  # введём цифру и выведем по 1й цифре (разложим)
kol = 0  # кол-во цифр
kol_ch = 0  # кол-во чётных цифр
summa = 0  # сумма всех цифр
pr = 1  # произведение всех цифр
maxi = 0  # максимальная из всех цифр (вводим самую маленькую цифру)
mini = 9  # минимальная из всех цифр (вводим самую большую цифру)
while x > 0:
    last = x % 10
    summa = summa + last
    pr = pr * last
    x = x // 10
    kol += 1
    if kol_ch % 2 == 0:
        kol_ch += 1
    if maxi < last:
        maxi = last
    if mini < last:
        mini = last
print("Вы ввели", kol, "цифр")
print("Вы ввели", kol_ch, "чётных цифр")
print("Сумма всех цифр равна ", summa)
print("Произведение всех цифр равно ", pr)
print("Максимальная цифра ", maxi)
print("Миниммальная цифра ", mini)
```

```python
# представим число например в 2-ичной системе
x = int(input())  # напр.14 = 8,4,2,1 -степени 2ки котрое учавствуют
# сложим степени 14 = 8(1)+4(1)+2(1)+1(0) 1=да,0=нет
while x > 0:
    last = x % 2
    print(last)  # вывод будет 1 1 1 0 в 2ичной системе только наоборот
    x = x // 2
```
