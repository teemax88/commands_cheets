```python
a = [0, 1, 2, 3, 4, 5, 2, 3, 4, 1, 1, 2, 3, 3, 5, 3, 2, 1]  # создали список со значениями от 0 до 5ти
count = [0, 0, 0, 0, 0, 0]  # или [0]*6. создали список с 6-ю нолями в который будем записывать кол-во 0,1 и т.д.
for i in a:  # в i присваиваем значение с переменной a
    count[i] += 1  # увеличим значение индекса в списке count,т.е. при появлении 4ки мы увеличим 4й индекс на 1
print(count)  # [1, 4, 4, 5, 2, 2] 1 раз 0, 4 раза 1 и т.д. вывод списка
for i in range(6):  # пройдём по циклу 6 раз поскольку в count 6 символов
    if count[i] != 0:  # напечетаем только те цифры которые есть
        print('Цифра ', i, 'встретилась ', count[i], 'раз')  # Цифра  0 встретилась  1 раз и т.д.
```

```python
# посчитаем сколько англ.букв встретится в строке
s = 'asfggGhgwjwheHGgUHSGDAHGFQWE 54422 /*--*&'
letters = [0] * 26  # создали пустой список (26 букв в англ.алфавите,регистр нам не важен)
for i in s.lower():  # пройдём по нашей переменной s (сделав буквы маленькими)
    if i >= 'a' and i <= 'z':
        nomer = ord(i) - 97  # присвоим в nomer число 0 для а, 1 для b и т.д.( ord('a')=97 )
        letters[nomer] += 1  # увеличим индекс найденной буквы на 1
for i in range(26):  # пройдём цикл 26 раз (столько может быть букв)
    if letters[i] > 0:  # если значение в списке не ноль
        print(chr(i + 97), letters[i],
              end='  ')  # chr приводит цифру машинного кода назад к числу, и сколько раз встретилась
        # a 2  d 1  e 2  f 2  g 8  h 5  j 1  q 1  s 2  u 1  w 3 вывод
```

## Построение матрицы из цифр
```python
rows = 3
cols = 3

matrix = []
for i in range(rows):
    row = []
    for j in range(cols):
        row.append(i * j)
    matrix.append(row)

for row in matrix:
    print(row)

[0, 0, 0]
[0, 1, 2]                                                  
[0, 2, 4]
```
