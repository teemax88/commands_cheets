```python
# 5 7 вводим сначала длинну списков 1 и 2
# 2 8 8 16 19 вводим список 1
# 3 4 5 5 10 11 12 вводим список 2
# 2 3 4 5 5 8 8 10 11 12 16 19 должны получить такой список
n, m = 5, 7  # n,m = map(int, input().split()) если вводим с клавы
a = [2, 8, 8, 16, 19]  # a = list(map(int, input().split())) если вводим с клавы
b = [3, 4, 5, 5, 10, 11, 12]  # b = list(map(int, input().split())) если вводим с клавы
i = j = 0  # создали указатели индекса в списке
c = []  # создали пустой итоговый список
while i < n and j < m:  # пока указатели меньше чем длинна списка
    if a[i] < b[j]:  # переменная а с индексом[0] меньше b[0] ?
        c.append(a[i])  # добавили в переменную с значение a[i] (меньшее)
        i += 1  # сдвинули индекс меньшего на 1 выше
    else:
        c.append(b[j])  # добавили в переменную с значение b[j] (меньшее)
        j += 1  # сдвинули индекс меньшего на 1 выше
while i < n:  # если в списке (a) что-то осталось то запишем в (с)
    c.append(a[i])  # добавили в переменную с значение a[i] (меньшее)
    i += 1  # сдвинули индекс меньшего на 1 выше
while j < m:  # если в списке (b) что-то осталось то запишем в (с)
    c.append(b[j])  # добавили в переменную с значение b[j] (меньшее)
    j += 1  # сдвинули индекс меньшего на 1 выше
# print(c) печаем новый список так [2, 3, 4, 5, 5, 8, 8, 10, 11, 12, 16, 19]
print(*c)  # распакует список т.е будут отдельно цифры через пробел и без скобок = 2 3 4 5 5 8 8 10 11 12 16 19
# или так можно сделать print(*c) так :
```
