# квадратная таблица(матрица) N*N (3*3 например)
# главная диагональ с координатами i=j [0][0] , [1][1] , [2][2].разрезает матрицу на верхн. и нижний треугольник
# у верхнего треугольника номер строки меньше номера столбца i < j ,у нижнего наоборот

a = []  # создадим квадр.матрицу в которой в диагональ значение 10,в верхний треуг.=5,нижн.=3
n = int(input())  # кол-во строк и столбцов
for i in range(n):
    a.append([0] * n)  # заполнит список нулями
for i in range(n):
    for j in range(n):
        if i == j:
            a[i][j] = 10
        elif i < j:
            a[i][j] = 5
        else:
            a[i][j] = 3
for i in a:  # выведет получившийся список
    print(i)
