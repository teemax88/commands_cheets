```python
# В данном уроке вспомним области видимости функций.
# Создадим функцию, внутри которой будут объявлены вложенные функции
# Поговорим о локальных, глобальных и нелокальных переменных
# ПРИОРИТЕТ В ПОИСКЕ ПЕРЕМЕННОЙ ОТДАЁТСЯ ЛОКАЛЬНОЙ ОБЛАСТИ, т.е если будет с одинак.названием в глобальной области
# и в локальной, то приоритет будет локальной

g = 'Grey global'  # глобальная переменная


def Colors():
    y = 'Yellow'  # локальная переменная которая будет видна только внутри своей функции и вложенным функциям
    g = 'Green local'  # локальная переменная

    def print_red():
        global g  # скажет что g нужно взять глобальную
        nonlocal y  # изменит y на 'New Yellow' не локально, т.е изменит во всей области,но сначала выведет 'Yellow'
        # так как изменения на 'New Yellow' были после print(r, y, g)
        r = 'Red'  # создали локальную переменную которая будет видна только внутри своей функции
        print(r, y, g)
        y = 'New Yellow'

    def print_blue():
        b = 'Blue'  # создали локальную переменную которая будет видна только внутри своей функции
        print(b, y, g)

    # print(r, b) не выведет поскольку r,b локальные переменные только в области видимости своих функций
    print_red()
    print_blue()


# print_red() так вызвать функцию нельзя поскольку она локальная, а не глобальная
Colors()  # Red Yellow Grey global


# Blue New Yellow Green local

# можно вызвать подобную функцию и она будет делать то что прописано в параметре, например...
def Colors_1(color='r'):  # если ничего не передать, то примет за 'r'
    def print_red():
        r = 'Red'
        print(r)

    def print_blue():
        b = 'Blue'
        print(b)

    if color == 'r':
        print_red()
    elif color == 'b':
        print_blue()
    else:
        print("I don't know this color ")


Colors_1()  # Red
Colors_1('b')  # Blue
Colors_1('asd')  # I don't know this color

```
