```python
"""
Функция map принимает два аргумента
1) функцию 2) итерируемый объект (коллекция) Т.Е. map(func,object)
Map возвращает итератор, в котором к каждому объекту коллекции применяется переданная функция.
"""

a = [-1, 2, -3, 4, -5]  # есть список а
b = list(map(abs, a))  # map вызовет abs для каждого элемента и запишет действие в список который присвоили переменной b
c = [abs(i) for i in a]  # делает тоже самое что и map выше
print(b)  # [1, 2, 3, 4, 5]
print(c)  # [1, 2, 3, 4, 5]
```

```python
def square(x):
    return x ** 2

q = [-1, 2, -5, 7, -8]
w = list(map(square, q))  # map вызовет square для каждого элем. q и запишет действие в список который присвоили пер.w
print(w)  # [1, 4, 25, 49, 64] # вывод положительных значений
```

```python
# можно в функции map применять функции и методы для строк, а также анонимные функции как lambda
e = ['heLlo', 'my', 'FRienD']
e_1 = list(map(len, e))  # запись в е_1 длины наших элементов
e_2 = list(map(str.upper, e))  # запись в е_2 наших элементов с большой буквы
e_3 = list(map(lambda x: x[::-1], e_2))  # функция lambda в х подставит поочерёдно элемен.(слова) перевернув их наоборот
e_31 = [i[::-1] for i in e_2]  # сделает тоже самое что и выше через генератор списков
e_4 = list(map(lambda x: x + '!', e_2))  # функция lambda в х подставит поочерёдно элемен.(слова) слово+!
e_5 = list(map(list, e_2))  # запишет каждое слово в список сост. из отдельных букв(каждая строка разобьётся на список)
e_51 = list(map(sorted, e_5))  # отсортирует предидущий список в алфавитном пор. для каждого списка
print(e_1)  # [5, 2, 6]
print(e_2)  # ['HELLO', 'MY', 'FRIEND']
print(e_3)  # ['OLLEH', 'YM', 'DNEIRF']
print(e_31)  # ['OLLEH', 'YM', 'DNEIRF']
print(e_4)  # ['HELLO!', 'MY!', 'FRIEND!']
print(e_5)  # [['H', 'E', 'L', 'L', 'O'], ['M', 'Y'], ['F', 'R', 'I', 'E', 'N', 'D']]
print(e_51)  # [['E', 'H', 'L', 'L', 'O'], ['M', 'Y'], ['D', 'E', 'F', 'I', 'N', 'R']]
```
