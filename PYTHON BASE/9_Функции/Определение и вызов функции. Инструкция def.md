```python
# определение функции (создали функцию с именем SayHello и с её инструкциями (отступы являются определением кол-ва)
# сначала нужно определить функцию а потом к ней обращаться
def SayHello():
    print('hello')
    print('world')
    print('and everybody')


def Square(x):  # вместо x мы подставим значение когда будем вызывать нашу функцию (вызов функции только с 1м х)
    print('Квадрат числа', x, 'будет равен', x ** 2)


def Multirly(x, y):  # вместо x,y мы подставим значения когда будем вызывать нашу функцию(вызов только с 2-мя x,y)
    print('Произведение чисел', x, '*', y, 'равно', x * y)


def even(x):  # функция проверяет введёное число на чётность
    if x % 2 == 0:
        print(x, 'это чётное число')
    else:
        print(x, 'это нечётное число')


def Factorial(x):  # функция считает факториал от числа
    pr = 1
    for i in range(2, x + 1):
        pr *= i
    print(pr)


i = int(input('Введите число '))  # функция Primer вызовется при разных условиях
if i >= 0:
    def Primer():
        print('Положительное')
else:
    def Primer():
        print('Отрицательное')


def Primer_2(i):  # функция Primer_2 вызовется и будет работатать в зависимости от условия
    if i >= 0:
        print('введённое число положительное')
    else:
        print('введённое число отрицательное')


# основная программа
SayHello()  # вызов фукнции hello world and everybody
Square(5)  # вызов фукнции Квадрат числа 5 будет равен 25
for i in range(1, 5):
    Square(i)  # Квадрат числа 1 будет равен 1 Квадрат числа 2 будет равен 4 и т.д.
Multirly(3, 8)  # Произведение чисел 3 * 8 равно 24
for i in range(10, 15):
    even(i)  # 10 это чётное число 11 это нечётное число 12 это чётное число и т.д.
Factorial(5)  # 120 функция посчитала факториал от 5-и
Primer()  # Положительное (при условии, что введётся положительное число)
Primer_2(23)  # введённое число положительное

```
