# set множество {} убирает все дубли

a = {9, 5, 3, 1, 3, 45, 1, 3}
print(a)  # {1, 3, 5, 9, 45} вывод элементов (произвольно, неотсортированных) без дублей

a1 = {"asd", 'asd', 'qwe', 'qwe', 'fg'}
print(a1)  # {'fg', 'qwe', 'asd'} вывод элементов без дублей (неотсортированных в произвольном порядке)

c = set('asdfasdfqw')  # set разбивает строку на символы и запишет множество из недублей
print(c)  # {'s', 'q', 'w', 'f', 'a', 'd'} вывод символов в случайном порядке без дублей

c1 = set([1, 3, 2, 9, 1, 3, 7, 2])
print(c1)  # {1, 2, 3, 7, 9} вывод символов в сортированном порядке без дублей
# c2 = set([[1, 3], [2, 9], [1, 3, 7, 2]])
# print(c2)  # TypeError: unhashable type: 'list' set может принимать только неизменяемые обьекты

d = set(range(5))
d1 = set()  # создали пустое множество
print(d)  # {0, 1, 2, 3, 4} вывод символов в сортированном порядке от 0 до 5

# уберём со списка повторы
a = [1, 8, 7, 1, 3, 45, 1, 3]
a = list(set(a))  # в а запишет результат множества от списка а(set) (уберёт повторы) преобразовав его в список (list)
print(a)  # [1, 3, 7, 8, 45] вывод отсортированного списка

# добавление к множеству
a = {1, 3, 45, 2, 7}
a = sorted(a)  # отсортирует множество по возрастанию
a = sorted(a, reverse=True)  # отсортирует множество по убыванию
a.add(23)  # добавит элемент если его не было
a.add(2)  # недобавит элемент если он уже есть
a.update([33, 3, 9])  # добавление итерабельных объектов (добавит элементы если их нет, если есть то ничего)
a.update('asddsa')  # добавление итерабельных объектов (добавит элементы(буквы) если их нет, если есть то ничего)
a.update(range(100, 103))  # добавление итерабельных объектов (добавит элементы если их нет, если есть то ничего)
a.update({77, 3, 9})  # добавление итерабельных объектов (добавит элементы если их нет, если есть то ничего)
print(a)  # {1, 2, 3, 7, 9, 77, 23, 's', 33, 'a', 100, 101, 102, 45, 'd'} вывод множества а (произвольно)

# удаление из множества
a = {1, 3, 45, 2, 7}
a.discard(3)  # удалит элемент 3 (если сделать повтор то ошибки не будет т.е. удаляет элемент если он есть)
a.remove(7)  # удалит элемент 3 (если сделать повтор то будет ошибка т.е. удаляет элемент только если он есть)
a.pop()  # удалит случайный элемент (если множество пустое то ошибка)
a.clear()  # очистит список
print(a)  # {2, 45} после clear = set()

# операции над множеством
a = {1, 3, 45, 3, 2, 2, 7}
print(len(a))  # 5 кол-во элементов (без повторов)
print(5 in a, 3 not in a)  # False False проверит есть ли элемент во множестве

a = {4, 3, 2, 1}
b = {3, 4, 5, 6}
print(a & b)  # {3, 4} символ & найдёт между множествами a и b элементы которые есть в обоих переменных
print(a.intersection(b))  # так же как & , найдёт между множествами a и b элементы которые есть в обоих переменных
a &= b  # в а присвоит пересечение а и b {3, 4}
a.intersection_update(b)  # сделает тоже самое что &= выше, в а присвоит пересечение а и b {3, 4}
print(a, b)  # {3, 4} {3, 4, 5, 6}
print(a | b)  # {3, 4, 5, 6} выведет объединение множеств без повторов
print(a.union(b))  # {3, 4, 5, 6} тоже самое что и (a | b) (просто покажет)
a = a.union(b)  # или (a |= b) {3, 4, 5, 6} запишет в а объдинение а и b

a = {4, 3, 2, 1}
b = {3, 4, 5, 6}
print(a - b)  # {1, 2} покажет только те элементы которые останутся после удаления общих
# a -= b # запишет в а только те элементы которые останутся после удаления общих
print(b - a)  # {5, 6} покажет только те элементы которые останутся после удаления общих
# b -= a # запишет в b только те элементы которые останутся после удаления общих

a = {4, 3, 2, 1}
b = {3, 4, 5, 6}
print(a ^ b)  # {1, 2, 5, 6} покажет те элементы которые небыли общими
# a ^= b # {1, 2, 5, 6} запишет в а те элементы которые небыли общими

a = {4, 3, 2, 1}
b = {3, 4, 2, 1, 2, 1}
print(a == b)  # True множества будут равны если их длина и их элементы будут равны (без повторов)

a = {4, 3, 2, 1}
b = {1, 2, 3}
c = {4, 5, 6}
print(a < b)  # False
print(a > b)  # True проверяет если во множ.а есть все элементы cо множ.b и что-то ёщё
print(a > c)  # False

a = {4, 32, 27, 1}

for i in a:
    print(i)  # вывод 32 1 27 4 только так можно обойти элементы множества

# пример запишем и выведем сколько было новых слов

    a.update(slova)  # добавим во множество slova без повторов
    text = input('Работает пока что-то вводим, STOP когда пусто+Enter ')
print(len(a), a)  # вывод длины множества и его самого, ниже пример ввода-вывода
