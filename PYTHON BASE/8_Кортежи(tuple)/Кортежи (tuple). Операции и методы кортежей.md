```python
# кортеж(tuple) неизменяемая  последовательность ''строка []список ()кортеж {} словарь
a = (1, 2, 3)
b = 1, 2, 3
b = 1,  # кортеж с одним элементом
c = tuple(range(1, 4))  # после tuple может быть итерабельная последовательность
d = tuple([1, 2, 3])
e = tuple((1, 2, 3))
print(a, b, c, d, e)  # вывод(запись) идинаковая (1, 2, 3) (1, 2, 3) (1, 2, 3) (1, 2, 3) (1, 2, 3)
a1 = ()  # или a1 = tuple() создали пустой кортеж
len(a)  # длина кортежа 3
print(2 in a, 5 in a)  # найдёт есть ли 2 в кортеже а True, 5ки нет поэтому False
b = 6, 7, 9
print(a + b, b + a)  # (1, 2, 3, 6, 7, 9) (6, 7, 9, 1, 2, 3) сцепление кортежей(сложение)
print(a * 2)  # (1, 2, 3, 1, 2, 3) дублирование кортежей
print(min(a), max(a))  # 1 3 вернёт миним.имаксим. значение если в кортеже используются числа
print(sum(a))  # 6 сумма элементов кортежа а
a = (1, 3, 5, True, 'asd', [10, 15, 20])
print(a[4])  # asd вывод
print(a.index('asd'))  # 4 индекс найденного элемента в кортеже
print(a.count(5))  # 1 посчитает сколько раз встретилась 5ка
a[5].append(40)  # добавит значение 40 в список (хотя кортеж не изменяется, а список можно изменить)
print(a)  # (1, 3, 5, True, 'asd', [10, 15, 20, 40])

q = (1, 2, 3)
q1 = list(q)  # преобразование кортежа в список
print(q1)  # [1, 2, 3]
print(q.__sizeof__(), q1.__sizeof__())  # 48 64 покажет размер нашего кортежа и списка (кортеж меньше)
w = {}
for i in q:  # запишет в словарь w ключи из q, а значения 'hello'
    w[i] = 'hello'
print(w)  # {1: 'hello', 2: 'hello', 3: 'hello'}

# можно кортеж преобразовать в список сделать в нём необходимые операции и снова преобразовать его в кортеж
l = 12, 34, 'asd'
l1 = list(l)
l1.append(345)
l1.reverse()
l = tuple(l1)
print(l)  # (345, 'asd', 34, 12)

tuple_ = ('aa', 'bb', 'cc')  # a tuple iterable
print(' = '.join(tuple_))  # join with the ' = ' separator
# aa = bb = cc
```
