```python
def myfunc(c):
    # комментарий не влияет на первую строку поскольку Python его не видит
    # если поставить """enter""" то программа сама предложит :param c:  и :return:
    """
    Наша функция которая ничего не далает, но в ней можно описать что
    она должна делать и это увидеть в help или .__doc__
    :param c:
    :return:
    """
    a = 12 * 123
    b = (23 * a) / c
    return b

```
```python
help(myfunc)  # если нет первой строки,то ничего не будет.Если есть то обычно это описание функции
print(myfunc.__doc__)  # Наша функция которая ни... (отображается только 1-я строчка после комментария)
```
