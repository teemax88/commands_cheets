## Округление
```python
import math

a = math.trunc(32.3)
print(a)    # ответ 32 ##trunc отсекает дробную часть числа
# (в функции может быть только 1 число)
```

```python
b = int(32.234)
print(b)    # ответ 32 ##тоже самое что trunc только без импорта math(приводит число к int)
```

```python
c = math.floor(32.987)
print(c)    # ответ 32 (с отрицанием следующее меньшее целое число) ##floor округление вниз
```

```python
c = math.floor(-32.987)
print(c)    # ответ -33
# math.floor (5) будет 5 поскольку 5 уже целое ближайшее меньшее число
```

```python
s = math.ceil(3.234)
print(s)    # ceil округление в большую сторону (ответ 4)
```

```python
s = math.ceil(-3.234)
print(s)    # ceil округление в большую сторону (ответ -3)
```

```python
# например
print(math.ceil(10 / 3))  # ответ 4 (10/3=3.33 в большую сторону=4)
print(math.floor(10 / 3))  # ответ 3 (10/3=3.33 в меньшую сторону=3)
print(math.trunc(10 / 3))  # ответ 3 (10/3=3.33 отсекли дробную часть=3)
```

## F-строки
```python
import datetime

today = datetime.datetime.today()
f"{today:%Y-%m-%d}"  # 2022-03-11
f"{today:%Y}"    # 2022
```

```python
x = 10
f"{x = :.3f}"    # x = 10.000
```

```python
text = "hello world"
# Центрирование текста:
f"{text:^15}"    # '  hello world  '

number = 1234567890
# Установка разделителя групп разрядов
f"{number:,}"    # 1,234,567,890

number = 123
# Добавление начальных нулей
f"{number:08}"   # 00000123
```

```python
number = 254.3463
f"{f'${number:.3f}':>10s}"   # '  $254.346'

num = 2.3123
f'{num:.2f}' # Limit to 2 decimals # 2.31
```

```python
'{:<30}'.format('left aligned')
'left aligned                  '
'{:>30}'.format('right aligned')
'                 right aligned'
'{:^30}'.format('centered')
'           centered           '
'{:*^30}'.format('centered')  # use '*' as a fill char
'***********centered***********'
```

```python
# format also supports binary numbers
"int: {0:d};  hex: {0:x};  oct: {0:o};  bin: {0:b}".format(42)
'int: 42;  hex: 2a;  oct: 52;  bin: 101010'

# with 0x, 0o, or 0b as prefix:
"int: {0:d};  hex: {0:#x};  oct: {0:#o};  bin: {0:#b}".format(42)
'int: 42;  hex: 0x2a;  oct: 0o52;  bin: 0b101010'
```

```python
'{:,}'.format(1234567890)
'1,234,567,890'
```

## Нахождение всех делителей числа
```python
n = int(input())  # найдём все делители т.е. 20= все числа которые делятся без остатка
# 1,2,4,5,10,20 всегда самое малелькое 1 и самое большое само число 20
i = 1
while i <= n:
    if n % i == 0:
        print(i, end=' ')  # отменим перенос строки, вывод будет в 1у строчку
    i += 1
```

```python
n = int(input())  # так программа работает быстрее поскольку она не ищет в промежутке от самого числа до //2
i = 1
while i <= n // 2:
    if n % i == 0:
        print(i, end=' ')  # отменим перенос строки, вывод будет в 1у строчку
    i += 1
print(n)  # вывод самого числа поскольку мы его не искали в n // 2
```

```python
n = int(input())  # можно все 3-и n, i, a = int(input()), 1, []
i = 1  # минимальный делитель любого числа
a = []  # создали пустой список в который запишем по очереди все делители
while i * i <= n:  # середина делителей [1, 2, 4, 5, 10, 20, 25, 50, 100] т.е. 10*10 <= 100
    # поскольку (1*100 = 2*50 = 4*25 = 5*20 = 10*10 )условие до середины
    if n % i == 0:  # если нашли первый делитель
        a.append(i)  # добавили в список (а) найденный делитель
        if i != n // i:  # в первом случае если 1 != 100//1,не сработает только на середине если 10 != 100//10 и не запишет повтор
            a.append(n // i)  # добавили в список (а) найденный ответный делитель (i=1 , 100//i=100)
    i += 1
a.sort()  # отсортировали наш список по возрастанию
print(a)
# print(sum(a)) например сумма делителей
```
